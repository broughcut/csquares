  def sq(i=0.1,int=false)
    n = 6 + (3 * Math.log10(i).abs).ceil
    it = case i
      when 10 then @digits[0..3]
      when 5 then @digits[0..4]
      when 1 then @digits[0..6]
      else @digits[0..n]
    end
    result = delimit [it[0..3],*it[4..-1].chunk_into(3)]
    int ? result.gsub(/:/){}.to_i : result
  end
